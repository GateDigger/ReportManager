CREATE OR REPLACE FUNCTION "DUMMY_PLANNER"(in_exec_id IN "RPMR_RX"."EXECUTION_ID"%TYPE) RETURN "RPMR_RX"."EXECUTION_START_PL"%TYPE IS

    v_last_exec_start_pl "RPMR_RX"."EXECUTION_START_PL"%TYPE;

BEGIN
    
    SELECT RX."EXECUTION_START_PL" INTO v_last_exec_start_pl FROM "RPMR_RX" RX
    WHERE RX."EXECUTION_ID" = in_exec_id;

    
    RETURN GREATEST(v_last_exec_start_pl, SYSDATE) + 10.0 / 1440.0;

END "DUMMY_PLANNER";
